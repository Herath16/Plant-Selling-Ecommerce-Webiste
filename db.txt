create database plant_shop;
use plant_shop;
create table products(
	id INT AUTO_INCREMENT PRIMARY KEY,
    name varchar(100),
    price INT,
    imege_url varchar(200)
);
CREATE TABLE cart_items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    product_id INT,
    quantity INT,
    FOREIGN KEY (product_id) REFERENCES products(id)
);
